/*
 * poll for mouse input
 */
static unsigned int mouse_poll(struct file *file, poll_table * wait)
{
       poll_wait(file, &mouse.wait, wait);
       if (mouse.ready)
              return POLLIN | POLLRDNORM;
       return 0;
}
